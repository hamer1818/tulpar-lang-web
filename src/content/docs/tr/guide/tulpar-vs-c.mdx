```c
#include <stdio.h>
#include <string.h>
#include <ctype.h>

void trim(char *s) {
    int start = 0;
    while (isspace((unsigned char)s[start])) start++;

    int end = strlen(s) - 1;
    while (end >= start && isspace((unsigned char)s[end])) end--;

    int j = 0;
    for (int i = start; i <= end; i++) {
        s[j++] = s[i];
    }
    s[j] = '\0';
}

void lower_str(char *s) {
    for (int i = 0; s[i] != '\0'; i++) {
        s[i] = (char)tolower((unsigned char)s[i]);
    }
}

int main(void) {
    char giris[] = "  Tulpar DILI  ";

    trim(giris);
    lower_str(giris);

    char *ilk_kelime = strtok(giris, " ");

    printf("Temiz: %s\n", giris);
    printf("İlk kelime: %s\n", ilk_kelime);
    return 0;
}
```
