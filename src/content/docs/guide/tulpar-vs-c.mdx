## String Processing

<TulparTabs>
  <TulparTabItem label="Tulpar" icon="tulpar">

```tulpar

```

  </TulparTabItem>
  <TulparTabItem label="C Language">

```c
#include <stdio.h>
#include <string.h>
#include <ctype.h>

void trim(char *s) {
    int start = 0;
    while (isspace((unsigned char)s[start])) start++;

    int end = strlen(s) - 1;
    while (end >= start && isspace((unsigned char)s[end])) end--;

    int j = 0;
    for (int i = start; i <= end; i++) {
        s[j++] = s[i];
    }
    s[j] = '\0';
}

void lower_str(char *s) {
    for (int i = 0; s[i] != '\0'; i++) {
        s[i] = (char)tolower((unsigned char)s[i]);
    }
}

int main(void) {
    char input[] = "  TULPAR LANG  ";

    trim(input);
    lower_str(input);

    char *first_word = strtok(input, " ");

    printf("Cleaned: %s\n", input);
    printf("First word: %s\n", first_word);
    return 0;
}
```

## Input and Error Handling


  </TulparTabItem>
  <TulparTabItem label="C Language">

```c
#include <stdio.h>

int main(void) {
    int age;
    printf("Enter your age: ");

    if (scanf("%d", &age) != 1) {
        printf("Invalid input.\n");
        return 1;
    }

    if (age < 18) {
        printf("Access denied.\n");
    } else {
        printf("Welcome!\n");
    }

    return 0;
}
```
